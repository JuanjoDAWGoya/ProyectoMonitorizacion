// popup.js // Obtener el historial de URLs visitadas desde chrome.storage chrome.storage.local.get('visitedUrls', function(result) { const visitedUrls = result.visitedUrls || []; const urlList = document.getElementById('urlList'); // Si hay URLs visitadas, las mostramos if (visitedUrls.length > 0) { visitedUrls.forEach(function(item) { const listItem = document.createElement('li'); listItem.textContent = `URL: ${item.url} | Usuario: ${item.user} | Fecha: ${item.timestamp}`; urlList.appendChild(listItem); }); } else { const listItem = document.createElement('li'); listItem.textContent = 'No se han registrado URLs visitadas.'; urlList.appendChild(listItem); } });
